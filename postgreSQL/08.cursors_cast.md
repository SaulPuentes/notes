CREATE OR REPLACE FUNCTION integer_to_boolean(SMALLINT)
	RETURNS BOOLEAN 
	LANGUAGE plpgsql AS
	$$
	BEGIN
		IF ($1 % 2 > 0) THEN
			RETURN FALSE;
		ELSE
			RETURN TRUE;
		END IF;
	END;
	$$;
